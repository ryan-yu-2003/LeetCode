class Solution:
    def candy(self, ratings: List[int]) -> int:
        '''
        時間複雜度O(n)
        空間複雜度O(n)
        '''
        n=len(ratings)
        candies = [1] * n
        #從左邊掃過去
        for i in range(1,n):
            if ratings[i]>ratings[i-1]:
                candies[i]=candies[i-1]+1
        #從右邊掃過去
        for i in range(n-2,-1,-1):
            if ratings[i]>ratings[i+1]:
                candies[i]=max(candies[i],candies[i+1]+1)

        
        return sum(candies)
