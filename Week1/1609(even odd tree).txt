# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: Optional[TreeNode]) -> bool:
        #BFS
        if not root:
            return True
        queue=deque([root])
        level=0
        while queue:
            a=len(queue)
            value=None
            for i in range(a):
                node=queue.popleft()
                if level%2==node.val%2:
                    return False
                else:
                    if value is not None:
                        if level%2==0 and node.val<=value:
                            return False
                        if level%2!=0 and value<=node.val:
                            return False
                    value=node.val
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
            level+=1
        return True
