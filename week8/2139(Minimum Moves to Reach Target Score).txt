class Solution:
    def minMoves(self, target: int, maxDoubles: int) -> int:
        #逆推回去，從target走到1，遇到偶數就用double因為數字大用double效益最好，剛好又是逆著回去
        a=0 #記錄走幾步
        while target>1:
            if target%2!=0:
                target=target-1
                a+=1
            if maxDoubles!=0 and target%2==0:
                target=target//2
                a+=1
                maxDoubles-=1
            if maxDoubles==0:
                a=a+target-1
                break
        return a
