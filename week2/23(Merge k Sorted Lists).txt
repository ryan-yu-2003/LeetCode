# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        #divide and conquer
        if len(lists)==0:
            return None
        while len(lists)>1:
            array=[]
            for i in range(0,len(lists),2):
                List1=lists[i]
                if i+1<len(lists):
                    List2=lists[i+1]
                else:
                     List2=None
                merge=self.merge_two_list(List1,List2)
                array.append(merge)
            lists=array
        return lists[0]

            

    def merge_two_list(self,List1,List2):
        dummy=ListNode(-1)
        current=dummy
        while List1 and List2:
            if List1.val<=List2.val:
                current.next=List1
                List1=List1.next
            else:
                current.next=List2
                List2=List2.next
            current=current.next
        if List1:
            current.next=List1
        if List2:
            current.next=List2
        return dummy.next
